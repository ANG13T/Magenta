~~ Classic Bottles of Beer Song

task beerSong: Vacant() ->  
    String bottles;
    Integer i = 99;
    while [i >= 1] ->

        if [i == 1] ->
        bottles = "bottle";
        <-
        
        if [i > 1] ->
        bottles = "bottles";
        <-

        emit(i + " " + bottles + " of beer on the wall,");
        emit(i + " " + bottles + " of beer,");
        emit("Take one down, pass it around,");

        i = i - 1;
        emit(i + " bottles of beer on the wall!");
    <-
<-

Class Structure

  Integer age = 3;
  String name = "Bob";

    emit "Hello, my name is " + age;

Function Structure
